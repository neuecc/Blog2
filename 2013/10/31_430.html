Collection Framework

еүҚеҚҠйғЁеҲҶгҒ§гҒҷгҒҢгҖҒгҒ“гӮҢгҒ®гӮёгғҘгғјгғЁгғјгҒӘгҒЁгҒ“гӮҚгҒҜILinqable<T>гҖҒгҒҳгӮғгҒӘгҒҸгҒҰ(гӮ№гғ©гӮӨгғүиіҮж–ҷгҒ§гҒҜеҮәгҒҰгҒӘгҒ„гҒ®гҒ§гғҠгғігғҺгӮігғғгғҒгғЈгҒ§гҒҷгҒҢ)гҖҒReadOnlyзі»гҒ®еҸ–гӮҠжүұгҒ„гҒ§гҒҷгҒӯгҖӮMutableгҒЁReadOnlyгҒҢжһқеҲҶгҒӢгӮҢгҒ—гҒҰгҒ„гӮӢзҗҶз”ұгҒЁгҒӢеҜҫеҮҰж–№жі•гҒЁгҒӢгҖҒгҒҢдјқгҒҲгҒҹгҒӢгҒЈгҒҹзӮ№гҒ§гҒҷгҖӮгҒ„гӮ„гҒӮгҖҒгӮігғ¬гӮҜгӮ·гғ§гғідҪңгӮӢжҷӮгҒҜдёЎж–№е®ҹиЈ…гҒ—гӮҲгҒҶпјҒгҒЁгҒӢгҒ—гӮҮгҒҶгӮӮгҒӘгҒ„гҒ§гҒҷгҒӯгҒҲгҖҒshoganaiгҒ®гҒ§гҒҷгҒӯгҒҮвҖҰвҖҰгҖӮ

IEnumerable<T>гҒЁIReadOnlyCollection<T>гҒ®е·®з•°гҒҜе®ҹдҪ“еҢ–гҒ•гӮҢгҒҰгҒ„гҒӘгҒ„гҖҢеҸҜиғҪжҖ§гҒҢгҒӮгӮӢгҖҚгҒӢгҒ©гҒҶгҒӢгҖӮгҒ§гҖҒгҒӘгҒ®гҒ§гғЎгӮҪгғғгғүгҒ®еј•ж•°гҒӘгҒ©гҒ§еҶ…йғЁгҒ§е®ҹдҪ“еҢ–гҒ•гӮҢгҒҰгӮӢгҒ®гӮ’еүҚжҸҗгҒ«гҒ»гҒ’гӮӮгҒ’гҒ—гҒҹгҒ„е ҙеҗҲгҒҜгҖҒIReadOnlyCollection<T>гӮ’еҸ—гҒ‘еҸ–гӮӢгҒ»гҒҶгҒҢжңӣгҒҫгҒ—гҒ„гҒЁгҒ„гҒҲгҒ°жңӣгҒҫгҒ—гҒ„гҒ®гҒ§гҒҷгҒҢгҖҒжұҺз”Ёзҡ„гҒ«IEnumerableгҒ®гҒҫгҒҫгҒ§вҖҰвҖҰгҒЁгҒ„гҒҶе ҙеҗҲгҒҜгҖҒд»ҘдёӢгҒ®гӮҲгҒҶгҒӘгғЎгӮҪгғғгғүгӮ’з”Ёж„ҸгҒ—гҒЁгҒҸгҒЁгҒ„гҒ„гҒ§гҒ—гӮҮгҒҶгҖӮ

/// <summary>
/// sourceгҒҢйҒ…е»¶зҠ¶ж…ӢгҒ®е ҙеҗҲгҖҒе®ҹдҪ“еҢ–гҒ—гҒҰиҝ”гҒ—гҖҒж—ўгҒ«е®ҹдҪ“еҢ–гҒ•гӮҢгҒҰгҒ„гӮӢе ҙеҗҲгҒҜдҪ•гӮӮгҒӣгҒҡгҒқгӮҢиҮӘиә«гӮ’иҝ”гҒ—гҒҫгҒҷгҖӮ
/// </summary>
/// <param name="source">еҜҫиұЎгҒ®гӮ·гғјгӮұгғігӮ№гҖӮ</param>
/// <param name="nullToEmpty">trueгҒ®е ҙеҗҲгҖҒsourceгҒҢnullжҷӮгҒҜз©әгӮ·гғјгӮұгғігӮ№гӮ’иҝ”гҒ—гҒҫгҒҷгҖӮfalseгҒ®е ҙеҗҲгҒҜArgumentNullExceptionгӮ’еҗҗгҒҚгҒҫгҒҷгҖӮ</param>
public static IEnumerable<T> Materialize<T>(this IEnumerable<T> source, bool nullToEmpty = true)
{
    if (nullToEmpty && source == null)
    {
        return Enumerable.Empty<T>();
    }
    else
    {
        if (source == null) throw new ArgumentNullException("sourceгҒҢnullгҒ§гҒҷ");
    }

    if (source is ICollection<T>)
    {
        return source;
    }
    if (source is IReadOnlyCollection<T>)
    {
        return source;
    }

    return source.ToArray();
}

гҒ©гҒҶгҒ§гҒ—гӮҮгҖӮгҒҫгҒҹгҖҒдәҢеәҰиӘӯгҒҝгҒӘгӮүеҲ—жҢҷгҒ—гҒҹгӮүгӮӯгғЈгғғгӮ·гғҘгҒ—гҒҰгҖҒеҶҚеәҰиӘӯгӮҖжҷӮгҒҜгҒқгҒЈгҒӢгӮүиӘӯгӮ“гҒ§гҒҸгӮҢгӮҢгҒ°гҒ„гҒ„гҒ®гҒ«пјҒгҒЁгҒ„гҒҶгғӘгӮҜгӮЁгӮ№гғҲгҒӮгӮӢгҒӢгҒЁжҖқгҒ„гҒҫгҒҷгҖӮгҒқгӮҢгҒҜдёҖиҲ¬зҡ„гҒ«гҒҜгғЎгғўеҢ–(Memoization)гҒЁгҒ„гҒ„гҒҫгҒҷгҖӮгҒЁгҒ„гҒҶгӮҸгҒ‘гҒ§гҖҒгӮ·гғјгӮұгғігӮ№гҒ«е®ҹиЈ…гҒ—гҒҰгҒҝгҒҫгҒ—гӮҮгҒҶгҖӮ

public static IEnumerable<T> Memoize<T>(this IEnumerable<T> source)
{
    if (source == null) throw new ArgumentNullException("sourceгҒҢnull");
    return new MemoizedEnumerable<T>(source);
}

class MemoizedEnumerable<T> : IEnumerable<T>, IDisposable
{
    readonly IEnumerable<T> source;
    readonly List<T> cache = new List<T>();
    bool cacheComplete = false;
    IEnumerator<T> enumerator;

    public MemoizedEnumerable(IEnumerable<T> source)
    {
        this.source = source;
    }

    public IEnumerator<T> GetEnumerator()
    {
        if (enumerator == null) enumerator = source.GetEnumerator();
        return new Enumerator(this);
    }

    System.Collections.IEnumerator System.Collections.IEnumerable.GetEnumerator()
    {
        return GetEnumerator();
    }

    public void Dispose()
    {
        if (enumerator != null) enumerator.Dispose();
    }

    class Enumerator : IEnumerator<T>
    {
        readonly MemoizedEnumerable<T> enumerable;
        int index = 0;

        public Enumerator(MemoizedEnumerable<T> enumerable)
        {
            this.enumerable = enumerable;
        }

        public T Current { get; private set; }

        public void Dispose()
        {

        }

        object System.Collections.IEnumerator.Current
        {
            get { return Current; }
        }

        public bool MoveNext()
        {
            if (index < enumerable.cache.Count)
            {
                Current = enumerable.cache[index];
                index++;
                return true;
            }

            if (enumerable.cacheComplete) return false;

            if (enumerable.enumerator.MoveNext())
            {
                Current = enumerable.enumerator.Current;
                enumerable.cache.Add(Current);
                index++;
                return true;
            }

            enumerable.cacheComplete = true;
            enumerable.enumerator.Dispose();
            return false;
        }

        public void Reset()
        {
            throw new NotSupportedException("ResetгҒҜз”Је»ғ");
        }
    }
}

гҒӮгҒЁгҖҒIEnumerable<T>гҒ§гҒҷгҒҢгҖҒгӮ№гғ¬гғғгғүгӮ»гғјгғ•гҒ§гҒҜгҒӘгҒ„гҖӮгҒқгҒҶгҖҒIEnumerable<T>гҒ«гҒҜгӮ№гғ¬гғғгғүгӮ»гғјгғ•гҒ®дҝқиЁјгҒҜе®ҹгҒҜгҒӘгҒ„гҖӮгҒЁгҒ„гҒҶгҒ®гӮ’йҖҶжүӢгҒ«еҸ–гҒЈгҒҰгӮӢпјҲгҒҫгҒҒгҖҒгҒқгӮҢгҒҜгҒӮгӮ“гҒҫгӮҠгҒӘгҒ®гҒ§ж°—гҒ«гҒӘгӮӢдәәгҒҜlockгҒӢгҒ‘гҒҹгӮҠгҒ—гҒҫгҒ—гӮҮгҒҶпјүгҖӮгҒЎгҒӘгҒҝгҒ«ReadOnlyCollectionгҒ гҒЈгҒҰгғ©гғғгғ—е…ғгҒ®гӮ·гғјгӮұгғігӮ№гҒҢеӨүжӣҙгҒ•гӮҢгҒҹгӮүгӮ№гғ¬гғғгғүгӮ»гғјгғ•гҒҳгӮғгҒӘгҒ„гҖӮгҒқгҒ—гҒҰгҖҒгӮ№гғ¬гғғгғүгӮ»гғјгғ•жҖ§гҒҢе®Ңз’§гҒ«дҝқиЁјгҒ•гӮҢгҒҰгҒ„гӮӢгҒ®гҒҢImmutable CollectionsгҖӮгҒЁгҒ„гҒҶи©ұгҒ«гҒӨгҒӘгҒҢгҒЈгҒҹгӮҠгҒӨгҒӘгҒҢгӮүгҒӘгҒӢгҒЈгҒҹгӮҠгҖӮ

Immutable Collections

Immutable CollectionsгҒҜе®ҹиЈ…зҠ¶жіҒгҒҢ.NET Framework BlogгҒ§йҡҸжҷӮи§ҰгӮҢгӮүгӮҢгҒҰгҒ„гҒҰгҖҒгғӘгғӘгғјгӮ№жҷӮгҒ®Immutable collections ready for prime timeгӮ’иӘӯгӮҒгҒ°гҖҒгҒӘгӮ“гҒӘгҒ®гҒӢгҒЈгҒҰгҒҜеҲҶгҒӢгӮӢгҒ®гҒ§гҒҜгҒӢгҒЁгҖӮгҒқгҒ®дёҠгҒ§з§ҒгҒҢд»ҠеӣһгҒ§еүІгҒЁй…ёгҒЈгҒұгҒҸиЁҖгҒ„гҒҹгҒӢгҒЈгҒҹгҒ®гҒҜгҖҒReadOnlyгҖҢгҒ§гҒҜгҒӘгҒ„гҖҚгҒЈгҒҰгҒ“гҒЁгҒ§гҒҷгҖӮгҒқгҒ—гҒҰзөҗи«–гҒҜгӮўгғӘгӮӯгӮҝгғӘгҒ«дҪҝгҒ„еҲҶгҒ‘гӮҲгҒҶгҖҒгҒЁгҒ„гҒҶи©ұгҒ§гҒ—гҒҹгҖӮ
